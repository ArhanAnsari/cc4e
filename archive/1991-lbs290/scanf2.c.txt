
#include <stdio.h>

main() {

  char tmp[100];
  int i,j;

  int n;

  while (1) {

    printf("Calling gets ");

    if ( gets(tmp) == NULL ) break;

    printf("tmp=%s\n",tmp);

    if ( tmp[0] == '*' ) {
      printf("Got a comment!\n");
      continue;
    }

    n = sscanf(tmp,"%d %d",&i,&j);

    printf("n=%d i=%d j=%d\n",n,i,j);

  }

}
