Installation Notes
------------------


cd /root
git clone https://github.com/tsugiproject/tsugi-build.git
cd tsugi-build/docker/base
bash  tsugi-base-prepare.sh

apt-get install apt-transport-https ca-certificates curl gnupg lsb-release
apt-get update

# https://docs.docker.com/engine/install/ubuntu/
# https://stackoverflow.com/questions/60249177/e-the-repository-http-ppa-launchpad-net-certbot-certbot-ubuntu-focal-release

sudo apt-add-repository -r ppa:certbot/certbot

sudo apt update
sudo apt-get update

curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

echo \
  "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

apt-get update
apt-get install docker-ce docker-ce-cli containerd.io

docker run hello-world

cd /var/www
git clone https://github.com/csev/cc4e.git
replace html with cc4e
cd cc4e

# https://certbot.eff.org/lets-encrypt/ubuntufocal-apache

snap install core
snap refresh core
apt-get remove certbot
snap install --classic certbot
ln -s /snap/bin/certbot /usr/bin/certbot
certbot --apache

<Directory /var/www/>
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted
</Directory>
